# Wolf-Zero — Audio (procedural, autonomous) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Give the game sound with **zero external assets** — synthesize SFX in GDScript as `AudioStreamWAV`, wire them to existing gameplay signals, and add a simple procedural music bed. Placeholder-grade but real and fully self-contained (swap in CC0/real audio later).

**Architecture:** A static `SfxGenerator` builds `AudioStreamWAV`s from synthesized float samples (16-bit mono PCM). An `AudioManager` node owns an `AudioStreamPlayer` pool + a `name→stream` dict and a `play(name)` method; `main.gd` connects gameplay signals to it. A separate quiet looping `AudioStreamPlayer` plays a generated pentatonic "synth-koto" bed. Verification: unit tests on the generator/manager (valid streams, no crash); headless runs confirm wiring triggers without errors. **Actual listening is a human step** (headless has no audio device).

**Tech Stack:** Godot 4.6, GDScript, GUT. Binary `C:\Godot\Godot_v4.6.1-stable_win64.exe`. Tests `... -gtest=res://test/<f>.gd`; suite `-gconfig=res://.gutconfig.json`. New `class_name` → `--import` once.

- [ ] **Step 2: Implement `scripts/audio/sfx_generator.gd`:**
```gdscript
class_name SfxGenerator
extends RefCounted
## Procedural SFX: synthesize AudioStreamWAV (16-bit mono PCM) — no external files.

const RATE := 22050

## Build a stream from float samples in [-1, 1].
static func from_samples(samples: Array, rate: int = RATE) -> AudioStreamWAV:
	var bytes := PackedByteArray()
	bytes.resize(samples.size() * 2)
	for i in range(samples.size()):
		var v: float = clampf(samples[i], -1.0, 1.0)
		bytes.encode_s16(i * 2, int(v * 32767.0))
	var s := AudioStreamWAV.new()
	s.format = AudioStreamWAV.FORMAT_16_BITS
	s.stereo = false
	s.mix_rate = rate
	s.data = bytes
	return s

## Named SFX factory.
static func make(name: String) -> AudioStreamWAV:
	match name:
		"slash":   return from_samples(_noise_sweep(0.12, 1.0, 0.2, 0.7))      # bright noisy whoosh
		"hit":     return from_samples(_noise_burst(0.10, 0.9))                # crunchy impact
		"hit_light": return from_samples(_noise_burst(0.06, 0.5))
		"jump":    return from_samples(_blip(0.10, 320.0, 620.0, 0.5))         # rising blip
		"dash":    return from_samples(_noise_sweep(0.18, 0.6, 0.9, 0.3))      # airy sweep
		"dodge":   return from_samples(_blip(0.12, 500.0, 300.0, 0.4))         # falling whoosh-ish
		"parry":   return from_samples(_chime(0.30, 1200.0, 0.6))             # bright ring
		"echo":    return from_samples(_chime(0.40, 700.0, 0.45))            # ethereal tone
		"death":   return from_samples(_blip(0.45, 300.0, 80.0, 0.6))         # descending
		_:         return from_samples(_noise_burst(0.05, 0.4))
```
Add the helper synths (all return `Array` of floats; use a simple LCG so output is deterministic — Godot disallows `randf` here? randf is fine in normal scripts, but for determinism use a seeded value; randf is acceptable):
```gdscript
static func _env(i: int, n: int, attack: float) -> float:
	var t := float(i) / float(n)
	var a := minf(t / maxf(attack, 0.001), 1.0)      # attack ramp
	var d := 1.0 - t                                  # linear decay
	return a * d

static func _noise_burst(dur: float, amp: float) -> Array:
	var n := int(dur * RATE); var out := []; out.resize(n)
	for i in range(n):
		out[i] = (randf() * 2.0 - 1.0) * amp * _env(i, n, 0.01)
	return out

static func _noise_sweep(dur: float, amp: float, lp_start: float, lp_end: float) -> Array:
	# crude low-pass-ish by smoothing toward target; lp in [0,1]
	var n := int(dur * RATE); var out := []; out.resize(n); var prev := 0.0
	for i in range(n):
		var raw := (randf() * 2.0 - 1.0)
		var lp: float = lerpf(lp_start, lp_end, float(i) / float(n))
		prev = lerpf(prev, raw, lp)
		out[i] = prev * amp * _env(i, n, 0.02)
	return out

static func _blip(dur: float, f0: float, f1: float, amp: float) -> Array:
	var n := int(dur * RATE); var out := []; out.resize(n); var phase := 0.0
	for i in range(n):
		var f: float = lerpf(f0, f1, float(i) / float(n))
		phase += TAU * f / float(RATE)
		out[i] = sin(phase) * amp * _env(i, n, 0.01)
	return out

static func _chime(dur: float, f0: float, amp: float) -> Array:
	var n := int(dur * RATE); var out := []; out.resize(n)
	for i in range(n):
		var t := float(i) / float(RATE)
		var s := sin(TAU * f0 * t) * 0.6 + sin(TAU * f0 * 2.0 * t) * 0.3 + sin(TAU * f0 * 3.0 * t) * 0.1
		out[i] = s * amp * _env(i, n, 0.005)
	return out
```

- [ ] **Step 2: Implement `scripts/audio/audio_manager.gd`** (a `Node` with a small pool of `AudioStreamPlayer`s, round-robin):
```gdscript
class_name AudioManager
extends Node

const POOL := 8
var _sounds: Dictionary = {}           # name -> AudioStreamWAV
var _players: Array[AudioStreamPlayer] = []
var _next := 0
@export var sfx_db := -6.0

func _ready() -> void:
	process_mode = Node.PROCESS_MODE_ALWAYS
	for n in ["slash","hit","hit_light","jump","dash","dodge","parry","echo","death"]:
		_sounds[n] = SfxGenerator.make(n)
	for i in range(POOL):
		var p := AudioStreamPlayer.new()
		p.volume_db = sfx_db
		add_child(p)
		_players.append(p)

func sound_count() -> int:
	return _sounds.size()

func play(name: String) -> void:
	if not _sounds.has(name) or _players.is_empty():
		return
	var p := _players[_next]
	_next = (_next + 1) % _players.size()
	p.stream = _sounds[name]
	p.play()
```
